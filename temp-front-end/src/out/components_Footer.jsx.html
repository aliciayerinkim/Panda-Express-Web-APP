import React from "react";
import { useNavigate } from 'react-router-dom';
import { Button } from "react-bootstrap";

/**
 * Footer that allows any user to go back a page.
 * @author Joseph Quismorio
 */

function Footer() {
  const history = useNavigate();
  return (
    <div className="navigation" style={{paddingTop: 40}}>
      <nav className="navbar navbar-expand fixed-bottom navbar-dark bg-red">
        <div className="container">
          <Button variant="panda-btn-alt" size="md" onClick={() => history(-1)} style={{fontWeight: 800, color: '#D02B2E'}} className="my-2 ml-3">BACK TO MENU</Button>{' '}
        </div>
      </nav>
    </div>
  );
}

export default Footer;
